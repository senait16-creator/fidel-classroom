function switchMode() { isSignUpMode = !isSignUpMode; document.getElementById("authTitle").innerHTML = isSignUpMode ? "Welcome to the
Fidel Classroom!" : "Welcome Back"; document.getElementById("authSubtitle").innerText = isSignUpMode ? "Join your peers and master the Amharic Alphabet" : "Log into your profile"; document.getElementById("mainAuthBtn").innerText = isSignUpMode ? "Sign Up" : "Log In"; document.getElementById("switchAuthBtn").innerText = isSignUpMode ? "Already have a profile? Log In" : "New student? Create a profile"; } async function handleAuth() { const email = document.getElementById("email").value.trim(); const password = document.getElementById("password").value; if (!email || !password) return alert("Please enter both fields."); if (isSignUpMode) { const { data, error } = await _supabase.auth.signUp({ email, password }); if (error) return alert("Sign Up Error: " + error.message); if (data?.user) { currentUser = data.user; document.getElementById("studentEmail").innerText = currentUser.email; document.getElementById("authScreen").style.display = "none"; document.getElementById("dashboard").style.display = "none"; document.getElementById("profileSetupScreen").style.display = "block"; } } else { const { data, error } = await _supabase.auth.signInWithPassword({ email, password }); if (error) return alert("Login Error: " + error.message); if (data?.user) checkUserProfileStatus(data.user); } } async function checkUserProfileStatus(user) { currentUser = user; document.getElementById("studentEmail").innerText = user.email; const { data: profile, error } = await _supabase .from('profiles') .select('display_name, avatar_character') .eq('id', user.id) .maybeSingle(); if (profile && profile.display_name) { enterClassroomDashboard(profile.display_name, profile.avatar_character); } else { document.getElementById("authScreen").style.display = "none"; document.getElementById("dashboard").style.display = "none"; document.getElementById("profileSetupScreen").style.display = "block"; } } function selectAvatar(symbol, element) { selectedAvatarSymbol = symbol; document.querySelectorAll('.avatar-option').forEach(opt => opt.classList.remove('selected')); element.classList.add('selected'); } async function saveProfileData() { const nameInput = document.getElementById("displayName").value.trim(); if (!nameInput) return alert("Please provide a nickname or display name."); const { error } = await _supabase .from('profiles') .upsert({ id: currentUser.id, display_name: nameInput, avatar_character: selectedAvatarSymbol }); if (error) return alert("Error saving profile details: " + error.message); enterClassroomDashboard(nameInput, selectedAvatarSymbol); } async function enterClassroomDashboard(name, avatar) { document.getElementById("displayUserHeader").innerText = name; document.getElementById("displayAvatarHeader").innerText = avatar; document.getElementById("authScreen").style.display = "none"; document.getElementById("profileSetupScreen").style.display = "none"; document.getElementById("dashboard").style.display = "block"; await fetchUserProgress(); } async function fetchUserProgress() { if (!currentUser) return; const { data, error } = await _supabase .from('user_progress') .select('mastered_letters') .eq('user_id', currentUser.id) .maybeSingle(); if (data && data.mastered_letters) masteredLetters = data.mastered_letters; renderUIProgressUpdates(); } function renderUIProgressUpdates() { const cards = document.querySelectorAll('.fidel-card'); cards.forEach(card => { const letter = card.getAttribute('data-fidel'); if (masteredLetters.includes(letter)) card.classList.add('completed'); else card.classList.remove('completed'); }); const percent = cards.length > 0 ? Math.round((masteredLetters.length / cards.length) * 100) : 0; document.getElementById("progressBar").style.width = `${percent}%`; document.getElementById("progressText").innerText = `${percent}% Mastered`; } function openFidel(baseLetter, familyArray) { activeBaseFidel = baseLetter; document.getElementById("modalTitle").innerText = `"${baseLetter}" Family`; const vowelRow = document.getElementById("vowelRow"); vowelRow.innerHTML = ""; familyArray.forEach((char, index) => { const box = document.createElement('div'); box.className = 'vowel-box'; box.innerHTML = `${char}${standardVowels[index]}`; vowelRow.appendChild(box); }); const masterBtn = document.getElementById("masterBtn"); masterBtn.innerText = masteredLetters.includes(baseLetter) ? "Mark Family as Incomplete" : "Mark Family as Learned"; masterBtn.style.background = masteredLetters.includes(baseLetter) ? "#e74c3c" : "#2ecc71"; masterBtn.onclick = () => toggleLetterMastery(); document.getElementById("fidelModal").style.display = "flex"; } async function toggleLetterMastery() { if (masteredLetters.includes(activeBaseFidel)) masteredLetters = masteredLetters.filter(item => item !== activeBaseFidel); else masteredLetters.push(activeBaseFidel); const { error } = await _supabase.from('user_progress').upsert({ user_id: currentUser.id, mastered_letters: masteredLetters }, { onConflict: 'user_id' }); if (error) alert("Sync failed: " + error.message); else { renderUIProgressUpdates(); closeModal(null); } } function closeModal(event) { if (!event || event.target === document.getElementById("fidelModal")) { document.getElementById("fidelModal").style.display = "none"; activeBaseFidel = null; } } async function logout() { await _supabase.auth.signOut(); currentUser = null; masteredLetters = []; document.getElementById("dashboard").style.display = "none"; document.getElementById("profileSetupScreen").style.display = "none"; document.getElementById("authScreen").style.display = "block"; document.getElementById("email").value = ""; document.getElementById("password").value = ""; document.getElementById("displayName").value = ""; }
